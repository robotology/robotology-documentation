165 void EthReceiver::run()
166 {
167  ssize_t incoming_msg_size = 0;
168  ACE_INET_Addr sender_addr;
169  uint64_t incoming_msg_data[TheEthManager::maxRXpacketsize/8]; // 8-byte aligned local buffer for incoming packet: it must be able to accomodate max size of packet
170  const ssize_t incoming_msg_capacity = TheEthManager::maxRXpacketsize;
171 
172 #ifdef NETWORK_PERFORMANCE_BENCHMARK
173  m_perEvtVerifier.tick(yarp::os::Time::now());
174 #endif
175 
176 
177  int flags = 0;
178 #ifndef WIN32
179  flags |= MSG_DONTWAIT;
180 #endif
181 
182  static uint8_t earlyexit_prev = 0;
183  static uint8_t earlyexit_prevprev = 0;
184 
185  // marco.accame: set maxUDPpackets as a fixed minimum number (2) + the number of boards. all is multipled by rateofthread and by a gain which depends on past activity
186  // the gain on past activity is usually 1. if maxUDPpackets is not enough the gain becomes (1+f1). if again it is not enough, the gain becomes 1+f1+f2+f3 and stays as
187  // such until it is enough. at this time it becomes 1+f2 and then 1 again
188  const double f1 = 0.5;
189  const double f2 = 0.5;
190  const double f3 = 8.0;
191  double gain = 1.0 + f1*(1-earlyexit_prev) + f2*(1-earlyexit_prevprev) + f3*(1-earlyexit_prev)*(1-earlyexit_prevprev);
192  int maxUDPpackets = (2 + ethManager->getNumberOfResources()) * EthReceiver::rateofthread * gain;
193 
194 
195  earlyexit_prevprev = earlyexit_prev; // save previous early exit
196  earlyexit_prev = 0; // consider no early exit this time
197 
198  for(int i=0; i<maxUDPpackets; i++)
199  {
200  incoming_msg_size = recv_socket->recv((void *) incoming_msg_data, incoming_msg_capacity, sender_addr, flags);
201  if(incoming_msg_size <= 0)
202  { // marco.accame: i prefer using <= 0.
203  earlyexit_prev = 1; // yes, we have an early exit
204  break; // we break and do not return because we want to be sure to execute what is after the for() loop
205  }
206 
207  // we have a packet ... we give it to the ethmanager for it parsing
208  //bool collectStatistics = (statPrintInterval > 0) ? true : false;
209  ethManager->Reception(ethManager->toipv4addr(sender_addr), incoming_msg_data, incoming_msg_size);
210  }
211 
212  // execute the check on presence of all eth boards.
213  ethManager->CheckPresence();
214 }
