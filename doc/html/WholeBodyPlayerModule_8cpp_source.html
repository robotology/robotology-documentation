37  return false;

39 
40  // position move fallback

44  return false;
45  }
46 
47  // start
48  response.clear();
49  ok &= m_rpcPort.write(reqPlay, response);
50  if (!ok || response.get(0).asVocab32() != VOCAB_OK) {
