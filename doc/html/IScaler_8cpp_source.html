30 double IScaler::transform(double val) {
31  // update scaler if enabled
32  if(this->updateEnabled) {
33  this->update(val);
34  }
35  // check for division by zero
36  return (std::abs(this->scale) < 1.e-20) ? (val - this->offset) : (val - this->offset) / this->scale;
37 }
