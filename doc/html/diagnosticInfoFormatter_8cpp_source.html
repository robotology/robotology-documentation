38 
39  eOmn_info_extraformat_t extraf = static_cast<eOmn_info_extraformat_t>EOMN_INFO_PROPERTIES_FLAGS_get_extraformat(m_infobasic->properties.flags);
40  dnginfo.extraMessage.clear();
41 
42  if(eomn_info_extraformat_verbal == extraf)
43  {
44  dnginfo.extraMessage.append((NULL == m_extra) ? ("no extra info despite we are in verbal mode") : ((const char *)m_extra));

46  else
47  {
48  dnginfo.extraMessage.append(".");
49  }
50 
51 
52  ipv4ToString(dnginfo.baseInfo);
53  dnginfo.param64 = m_infobasic->properties.par64;
54  dnginfo.param16 = m_infobasic->properties.par16;
55  dnginfo.errorCode = m_infobasic->properties.code;
56 
57  EntityNameProvider entityNameProvider{dnginfo.sourceBoardIpAddr, m_ethManager};
58 
59  //2. create the parser related to the error cod family

66 
67  case eoerror_category_MotionControl: parser_ptr = std::make_unique<MotionControlParser>(dnginfo, entityNameProvider); break;
68 
69  case eoerror_category_HardWare: parser_ptr = std::make_unique<HwErrorParser>(dnginfo, entityNameProvider); break;

72

74 
75  case eoerror_category_Skin: parser_ptr = std::make_unique<SkinParser>(dnginfo, entityNameProvider); break;

77  case eoerror_category_InertialSensor: parser_ptr = std::make_unique<InertialSensorParser>(dnginfo, entityNameProvider); break;
78
