24 void Kalman::initialize()
25 {
26  n=A.rows();
27  m=H.rows();
28 
29  At=A.transposed();
30  Ht=H.transposed();
31  I=eye((int)n,(int)n);

33  x.resize(n,0.0);
34  P.resize(n,n); P.zero();

38 }

90 /**********************************************************************/
91 const Vector& Kalman::correct(const Vector &z)
92 {
93  Matrix invS=pinv(S);
94  K=P*Ht*invS;
95  Vector e=z-get_y();
96  x+=K*e;
97  P=(I-K*H)*P;
98  validationGate=yarp::math::dot(e,invS*e);
99  return x;
100 }
