114

117  // the context contains the tracking mode, the neck limits and so on.
118  igaze->storeContext(&startup_context_id);

121  igaze->setNeckTrajTime(0.8);

123

163  virtual void run()
164  {
165  lock_guard<mutex> lg(mtx);
166 
167  t=Time::now();
168  generateTarget();
169 
170  if (state==STATE_TRACK)
171  {
172  // look at the target (streaming)
173  igaze->lookAtFixationPoint(fp);
174 
175  // some verbosity
176  printStatus();
177 
178  // we collect from time to time
179  // some interesting points (POI)
180  // where to look at soon afterwards
181  storeInterestingPoint();
182 
183  if (t-t2>=SWITCH_STATE_PER)
184  {
185  // switch state
186  state=STATE_RECALL;
187  }
188  }
189 
190  if (state==STATE_RECALL)
191  {
192  // pick up the first POI
193  // and clear the list
194  Vector ang=poiList.front();
195  poiList.clear();
196 
197  fprintf(stdout,"Retrieving POI #0 ... (%s) [deg]\n",
198  ang.toString(3,3).c_str());
199 
200  // register the motion-done event, attaching the callback
201  // that will be executed as soon as the gaze is accomplished
202  igaze->registerEvent(*this);
203 
204  // look at the chosen POI
205  igaze->lookAtAbsAngles(ang);
206 
207  // switch state
208  state=STATE_WAIT;
209  }
210 
211  if (state==STATE_STILL)
212  {
213  // check if the torso movement is complete
214  bool done;
215  ipos->checkMotionDone(0,&done);
216  if (done)
217  {
218  fprintf(stdout,"done\n");
219 
220  t1=t2=t3=t;
221 
222  // switch state
223  state=STATE_TRACK;
224  }
225  }
226  }

242  void generateTarget()
243  {
244  // translational target part: a circular trajectory
245  // in the yz plane centered in [-0.5,0.0,0.3] with radius=0.1 m
246  // and frequency 0.1 Hz
247  fp[0]=-0.5;
248  fp[1]=+0.0+0.1*cos(2.0*M_PI*0.1*(t-t0));
249  fp[2]=+0.3+0.1*sin(2.0*M_PI*0.1*(t-t0));
250  }

253  {
254  if (t-t3>=STORE_POI_PER)
255  {
256  Vector ang;
257 
258  // we store the current azimuth, elevation
259  // and vergence wrt the absolute reference frame

263  igaze->getAngles(ang);
264 
265  int numPOI=(int)poiList.size();
266  fprintf(stdout,"Storing POI #%d: (%s) [deg]\n",
267  numPOI,ang.toString(3,3).c_str());
268 
269  poiList.push_back(ang);
270 
271  t3=t;
272  }
