128 void simFaceExp::setSubSystem(const char *command)
129 {
130  //variable holding position in the array
131  int pos = 0;
132  //get sub system id and set writing position in the array
133  char systemID = command[0];
134  if (systemID == 'L') pos = 0;
135  else if (systemID == 'R') pos = 8;
136  //else if (systemID == 'M') pos = 16;
137  else if (systemID == 'M') pos = 21;
138 
139  //truncate string
140  char hex[2];
141  hex[0]=command[1];
142  hex[1]=command[2];
143 
144  //convert hexdec to binary
145  int num = strtol(hex, (char **)NULL, 16);
146 
147  if (systemID == 'S')
148  {
149  for (int i=0;i<8;i++)
150  eyeLids[i] = (num >> i)&1;
151 
152  //print the array
153  for (int i = 0; i<8;i++)
154  yDebug("%d", eyeLids[i]);
155  yDebug("\n");
156 
157  eyeLidPos = linearMap((float)num,36,72,0,30);
158  yDebug("eyeLid pos is %lf\n",eyeLidPos);
159  }
160  else
161  {
162  if (pos == 21)
163  {
164  for (int i=0;i<8;i++)
165  ledsActivation[(pos)-i] = (num >> i)&1;
166  }
167  else
168  for (int i=0;i<8;i++)
169  ledsActivation[pos+i] = (num >> i)&1;
170 
171  if (ledsActivation[19]==1 && ledsActivation[18]!=1)
172  {
173  ledsActivation[18]=1;
174  ledsActivation[19]=0;
175  }
176  else if (ledsActivation[18]==1 && ledsActivation[19]!=1)
177  {
178  ledsActivation[18]=0;
179  ledsActivation[19]=1;
180  }
181  }

213 float simFaceExp::linearMap( float x, float min, float max, float outMin, float outMax )
214 {
215  float m = - ( outMax-outMin )/( max-min );
216  float q = outMax - m*min;
217  float ret = m*x+q;
218  if (ret < outMin) return outMin;
219  if (ret > outMax) return outMax;
220  return ret;
221 }
