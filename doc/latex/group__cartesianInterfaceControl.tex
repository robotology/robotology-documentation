This module relies on \mbox{\hyperlink{icub_cartesian_interface}{The Cartesian Interface}} to provide a kind of easy-\/to-\/use port-\/based front-\/end for controlling the \mbox{\hyperlink{namespaceiCub}{i\+Cub}}\textquotesingle{}s arm in the operational space.\hypertarget{group__tutorial__perceptiveModels_lib_sec}{}\doxysubsection{Libraries}\label{group__tutorial__perceptiveModels_lib_sec}

\begin{DoxyItemize}
\item YARP libraries.
\end{DoxyItemize}\hypertarget{group__tutorial__perceptiveModels_parameters_sec}{}\doxysubsection{Parameters}\label{group__tutorial__perceptiveModels_parameters_sec}
--name {\itshape name} 
\begin{DoxyItemize}
\item The parameter {\itshape name} identifies the controller\textquotesingle{}s name; all the open ports will be tagged with the prefix /$<$name$>$/$<$part$>$/. If not specified {\itshape arm\+Ctrl} is assumed.
\end{DoxyItemize}

--robot {\itshape name} 
\begin{DoxyItemize}
\item The parameter {\itshape name} selects the robot name to connect to; if not specified {\itshape icub} is assumed.
\end{DoxyItemize}

--part {\itshape type} 
\begin{DoxyItemize}
\item The parameter {\itshape type} selects the robot\textquotesingle{}s arm to work with. It can be {\itshape right\+\_\+arm} or {\itshape left\+\_\+arm}; if not specified {\itshape right\+\_\+arm} is assumed.
\end{DoxyItemize}

--T {\itshape time} 
\begin{DoxyItemize}
\item specify the task execution time in seconds; by default {\itshape time} is 2.\+0 seconds. Note that this is just an approximation of execution time since there exists a controller running underneath.
\end{DoxyItemize}

--tol {\itshape tol} 
\begin{DoxyItemize}
\item specify the cartesian tolerance in meters for reaching the target.
\end{DoxyItemize}

--DOF8
\begin{DoxyItemize}
\item enable the control of torso yaw joint.
\end{DoxyItemize}

--DOF9
\begin{DoxyItemize}
\item enable the control of torso yaw/pitch joints.
\end{DoxyItemize}

--DOF10
\begin{DoxyItemize}
\item enable the control of torso yaw/roll/pitch joints.
\end{DoxyItemize}

--only\+XYZ
\begin{DoxyItemize}
\item disable orientation control.
\end{DoxyItemize}

--reinstate\+\_\+context
\begin{DoxyItemize}
\item reinstate controller context upon target reception.
\end{DoxyItemize}\hypertarget{group__tutorial__ationPrimitives_portsa_sec}{}\doxysubsection{Ports Accessed}\label{group__tutorial__ationPrimitives_portsa_sec}
